Copy raspberry pi image from one SD card to another
~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
sudo -s
lsblk   //list the devices
dd bs=4M if=/dev/mmcblk0 | gzip > /home/swathy/Documents/main_project/Energy_Internet/rpiImage.gz
gzip -dc /home/swathy/Documents/main_project/Energy_Internet/rpiImage.gz | dd bs=4M of=/dev/mmcblk0
